Verificar Item na Lista

Verificar Existencia de Item 
class Item{
    String descricao;
    double valor;
    
    Item({this.descricao,this.valor});
    
    List<Item> get buscarLista {
        return [
            new Item(descricao:'tenis nike',valor:299.00) ,
            new Item(descricao:'calca taquetel',valor:59.99) ,    
            new Item(descricao:'tenis',valor:19.00) ,    
        ];
    }
}

void main() {
    
    Item item = new Item();
    List<Item> lista = item.buscarLista;

    item.descricao = 'tenis';
    item.valor = 19.00;
    
    if ((lista.singleWhere((it) => it.descricao == item.descricao, orElse: () => null)) == null) {
        lista.add(item);
    }else{
        print('EXISTE !');
    }
    
}
